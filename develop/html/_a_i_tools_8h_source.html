115  template <typename EvalFun>
116  auto alphaBetaSearch(const State &state, const ActionState &actionState, gameModel::TeamSide mySide, double alpha,
117  double beta, int maxDepth, const EvalFun &evalFun, const std::atomic_bool &abort, unsigned long &expansions) ->
118  std::optional<std::pair<std::shared_ptr<gameController::Action>, double>> {
119  expansions++;
120  std::vector<std::shared_ptr<gameController::Action>> allActions;
121  auto currentPlayer = state.env->getPlayerById(actionState.id);
122  if(actionState.turnState == ActionState::TurnState::Action){
123  auto actionType = gameController::getPossibleBallActionType(currentPlayer, state.env);
124  if(!actionType.has_value()){
125  std::cerr << "-------No action possible-------" << std::endl;
126  return std::nullopt;
127  }
128 
129  if(*actionType == gameController::ActionType::Throw){
130  auto tmp = gameController::getAllPossibleShots(currentPlayer, state.env, 0);
131  for(const auto &a : tmp){
132  allActions.emplace_back(std::make_shared<gameController::Shot>(a));
133  }
134  } else {
135  auto chaser = std::dynamic_pointer_cast<gameModel::Chaser>(currentPlayer);
136  if(!chaser){
137  throw std::runtime_error("Player is no chaser");
138  }
139 
140  allActions.emplace_back(std::make_shared<gameController::WrestQuaffle>(gameController::WrestQuaffle(state.env, chaser, state.env->quaffle->position)));
141  }
142  } else {
143  auto tmp = gameController::getAllPossibleMoves(currentPlayer, state.env);
144  allActions.reserve(tmp.size());
145  for(const auto &a : tmp){
146  allActions.emplace_back(std::make_shared<gameController::Move>(a));
147  }
148  }
149 
150 
151  double minMaxVal = std::numeric_limits<double>::infinity();
152  std::optional<std::shared_ptr<gameController::Action>> minMaxAction;
153  bool maxSearch = gameLogic::conversions::idToSide(actionState.id) == mySide;
154  if(maxSearch){
155  minMaxVal *= -1;
156  }
157 
158  if(allActions.empty()){
159  std::cerr << "---------Action list is empty. Depth: " + std::to_string(maxDepth) << std::endl;
160  return std::nullopt;
161  }
162 
163  for(const auto &action : allActions){
164  double expectedValue = 0;
165  for(const auto &outcome : action->executeAll()){
166  const auto &currentEnv = outcome.first;
167  auto newState = state.clone();
168  newState.env = outcome.first;
169  newState.goalScoredThisRound = state.env->team1->score != newState.env->team1->score || state.env->team2->score != newState.env->team2->score;
170  auto playerOnSnitch = currentEnv->getPlayer(currentEnv->snitch->position);
171  if(abort || maxDepth == 0 || (playerOnSnitch.has_value() && INSTANCE_OF(*playerOnSnitch, gameModel::Seeker))){
172  return std::make_pair(action, evalFun(newState));
173  }
174 
175  double currentOutcomeExpectedValue = 0;
176  auto nextActors = computeNextActionStates(newState, actionState);
177  for(const auto &nextActor : nextActors){
178  auto tmp = alphaBetaSearch(nextActor.first, nextActor.second, mySide, alpha, beta, maxDepth - 1, evalFun, abort, expansions);
179  if(!tmp.has_value()){
180  return std::make_pair(action, evalFun(newState));
181  }
182 
183  currentOutcomeExpectedValue += tmp->second;
184  //Bedingung kann hier zutreffen, da abort von außen verändert werden kann
185  if(abort){
186  return std::make_pair(action, evalFun(newState));
187  }
188  }
189 
190  if(nextActors.empty()){
191  currentOutcomeExpectedValue = evalFun(newState);
192  } else {
193  currentOutcomeExpectedValue /= nextActors.size();
194  }
195 
196  expectedValue += currentOutcomeExpectedValue * outcome.second;
197  }
198 
199  if(maxSearch){
200  if(expectedValue > minMaxVal){
201  minMaxVal = expectedValue;
202  minMaxAction.emplace(action);
203  }
204 
205  if(expectedValue >= beta){
206  if(!minMaxAction.has_value()){
207  std::cerr << "----------error during pruning in max, no value. Depth: " + std::to_string(maxDepth) << std::endl;
208  return std::nullopt;
209  }
210 
211  return std::make_pair(minMaxAction.value(), minMaxVal);
212  }
213 
214  alpha = std::max(alpha, expectedValue);
215 
216  } else {
217  if(expectedValue < minMaxVal){
218  minMaxVal = expectedValue;
219  minMaxAction.emplace(action);
220  }
221 
222  if(expectedValue <= alpha){
223  if(!minMaxAction.has_value()){
224  std::cerr << "----------error during pruning in min, no value. Depth: " + std::to_string(maxDepth) << std::endl;
225  return std::nullopt;
226  }
227 
228  return std::make_pair(minMaxAction.value(), minMaxVal);
229  }
230 
231  beta = std::min(beta, expectedValue);
232  }
233  }
234 
235  if(!minMaxAction.has_value()){
236  std::cerr << "----------no value after all actions. Depth: " + std::to_string(maxDepth) << std::endl;
237  return std::nullopt;
238  }
239 
240  return std::make_pair(minMaxAction.value(), minMaxVal);
241  }

350  template <typename T, typename Exp, typename F>
351  auto aStarSearch(const SearchNode<T> &start, const SearchNode<T> &destination,
352  const Exp &expand, const F &f) -> std::optional<SearchNode<T>>{
353  std::deque<SearchNode<T>> visited;
354  std::list<SearchNode<T>> fringe = {start};
355  std::optional<SearchNode<T>> ret;
356  while(!fringe.empty()){
357  const auto currentNode = *fringe.begin();
358  fringe.pop_front();
359  if(currentNode == destination){
360  ret.emplace(currentNode);
361  break;
362  }
363 
364  bool nodeVisited = false;
365  for(const auto &node : visited){
366  if(currentNode == node){
367  nodeVisited = true;
368  break;
369  }
370  }
371 
372  if(nodeVisited){
373  continue;
374  }
375 
376  visited.emplace_back(currentNode);
377  for(const auto &newNode : expand(currentNode)){
378  bool inserted = false;
379  for(auto it = fringe.begin(); it != fringe.end(); ++it){
380  if(f(newNode) < f(*it)){
381  fringe.emplace(it, newNode);
382  inserted = true;
383  break;
384  }
385  }
386 
387  if(!inserted){
388  fringe.emplace_back(newNode);
389  fringe.size();
390  }
391  }
392  }
393 
394  return ret;
395  }

407  auto chooseBestAction(const std::vector<ActionType> &actionList, const EvalFun &evalFun, const bool &abort) ->
408  std::optional<std::tuple<typename std::vector<ActionType>::const_iterator, double>>{
409  std::optional<std::tuple<typename std::vector<ActionType>::const_iterator, double>> ret;
410  if(actionList.empty()){
411  throw std::runtime_error("List is empty. Cannot choose best entry");
412  }
413 
414  double highestScore = -std::numeric_limits<double>::infinity();
415  std::optional<typename std::vector<ActionType>::const_iterator> best;
416  for(auto action = actionList.begin(); action < actionList.end(); ++action){
417  if(abort){
418  break;
419  }
420 
421  double tmpScore = 0;
422  for(const auto &outcome : action->executeAll()){
423  tmpScore += outcome.second * evalFun(outcome.first);
424  }

427  highestScore = tmpScore;
428  best.emplace(action);
429  }
430  }
431 
432  if(best.has_value()){
433  ret.emplace(*best, highestScore);
434  }
435 
436  return ret;
