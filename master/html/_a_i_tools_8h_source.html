81  template <typename T, typename Exp, typename F>
82  auto aStarSearch(const SearchNode<T> &start, const SearchNode<T> &destination,
83  const Exp &expand, const F &f) -> std::optional<SearchNode<T>>{
84  std::deque<SearchNode<T>> visited;
85  std::list<SearchNode<T>> fringe = {start};
86  std::optional<SearchNode<T>> ret;
87  while(!fringe.empty()){
88  const auto currentNode = *fringe.begin();
89  fringe.pop_front();
90  if(currentNode == destination){
91  ret.emplace(currentNode);
92  break;
93  }
94 
95  bool nodeVisited = false;
96  for(const auto &node : visited){
97  if(currentNode == node){
98  nodeVisited = true;
99  break;
100  }
101  }
102 
103  if(nodeVisited){
104  continue;
105  }
106 
107  visited.emplace_back(currentNode);
108  for(const auto &newNode : expand(currentNode)){
109  bool inserted = false;
110  for(auto it = fringe.begin(); it != fringe.end(); ++it){
111  if(f(newNode) < f(*it)){
112  fringe.emplace(it, newNode);
113  inserted = true;
114  break;
115  }
116  }
117 
118  if(!inserted){
119  fringe.emplace_back(newNode);
120  fringe.size();
121  }
122  }
123  }
124 
125  return ret;
126  }

136  template <typename ActionType, typename EvalFun>
137  auto chooseBestAction(const std::vector<ActionType> &actionList, const EvalFun &evalFun) ->
138  std::tuple<typename std::vector<ActionType>::const_iterator, double>{
139  if(actionList.empty()){
140  throw std::runtime_error("List is empty. Cannot choose best entry");
141  }
142 
143  double highestScore = -std::numeric_limits<double>::infinity();
144  std::optional<typename std::vector<ActionType>::const_iterator> best;
145  for(auto action = actionList.begin(); action < actionList.end(); ++action){
146  double tmpScore = 0;
147  for(const auto &outcome : action->executeAll()){
148  tmpScore += outcome.second * evalFun(outcome.first);
149  }
150 
151  if(tmpScore > highestScore){
152  highestScore = tmpScore;
153  best.emplace(action);
154  }
155  }
156 
157  if(!best.has_value()){
158  throw std::runtime_error("No best candidate found");
159  }
160 
161  return {best.value(), highestScore};
162  }
